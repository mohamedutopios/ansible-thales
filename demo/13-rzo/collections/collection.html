Contenu pour le premier rôle : configure_interface

roles/configure_interface/tasks/main.yml

  • name: Configure network interface port2 using template
    fortinet.fortios.fortios_system_interface:
    vdom: "{{ vdom }}"
    state: present
    system_interface: "{{ lookup('template', 'interface_config.j2') }}"
    notify: Restart Network Interface

roles/configure_interface/vars/main.yml

vdom: "root"
interfaces:

  • name: "port2"
    ip: "10.0.0.1/24"
    allowaccess: "ping"
    alias: "WAN_Interface"

roles/configure_interface/templates/interface_config.j2

{
"name": "{{ interfaces[0].name }}",
"ip": "{{ interfaces[0].ip }}",
"allowaccess": "{{ interfaces[0].allowaccess }}",
"alias": "{{ interfaces[0].alias }}"
}

roles/configure_interface/handlers/main.yml

  • name: Restart Network Interface
    fortinet.fortios.fortios_monitor:
    vdom: "{{ vdom }}"
    selector: "dhcp-renew.system.interface"
    params:
    interface: "port2"

roles/configure_interface/meta/main.yml

dependencies: []

playbooks/playbook_interface.yml

  • name: Configure Fortigate Network Interface
    hosts: fortigates
    gather_facts: no
    connection: httpapi
    vars:
    ansible_network_os: fortinet.fortios.fortios
    ansible_httpapi_use_ssl: "no"
    ansible_httpapi_validate_certs: "no"
    ansible_httpapi_port: 80
    roles:
    • fortigate.demo.configure_interface

Contenu pour le deuxième rôle : configure_firewall

roles/configure_firewall/tasks/main.yml

  • name: Configure Firewall Policy using template
    fortinet.fortios.fortios_firewall_policy:
    vdom: "{{ vdom }}"
    state: present
    firewall_policy: "{{ lookup('template', 'firewall_policy.j2') }}"
    notify: Reload Firewall Policy

roles/configure_firewall/vars/main.yml

vdom: "root"
firewall_policies:

  • name: "Allow_HTTP"
    srcintf: "port2"
    dstintf: "port3"
    srcaddr: "all"
    dstaddr: "all"
    action: "accept"
    service: "HTTP"
    schedule: "always"
    logtraffic: "all"

roles/configure_firewall/templates/firewall_policy.j2

[
{
"name": "{{ firewall_policies[0].name }}",
"srcintf": ["{{ firewall_policies[0].srcintf }}"],
"dstintf": ["{{ firewall_policies[0].dstintf }}"],
"srcaddr": ["{{ firewall_policies[0].srcaddr }}"],
"dstaddr": ["{{ firewall_policies[0].dstaddr }}"],
"action": "{{ firewall_policies[0].action }}",
"service": ["{{ firewall_policies[0].service }}"],
"schedule": "{{ firewall_policies[0].schedule }}",
"logtraffic": "{{ firewall_policies[0].logtraffic }}"
}
]

roles/configure_firewall/handlers/main.yml

  • name: Reload Firewall Policy
    fortinet.fortios.fortios_monitor:
    vdom: "{{ vdom }}"
    selector: "reset.firewall.policy"

roles/configure_firewall/meta/main.yml

dependencies: []

playbooks/playbook_full_setup.yml

  • name: Full Fortigate Setup (Interface + Firewall)
    hosts: fortigates
    gather_facts: no
    connection: httpapi
    vars:
    ansible_network_os: fortinet.fortios.fortios
    ansible_httpapi_use_ssl: "no"
    ansible_httpapi_validate_certs: "no"
    ansible_httpapi_port: 80
    roles:
    • fortigate.demo.configure_interface
    • fortigate.demo.configure_firewall
